  456.   {

  459.     // create clusters
  460.     $clusters[] = new Cluster($center);
  461.   }
  462.   
  463.   // now deploy points to closest center
  464.   for ($a = 0; $a < 10; ++$a)         // 10 iterations is enough...
  465.   {       
  466.     foreach ($clusters as $cluster) {
  467.       $cluster->points = array();     // reinitialize points
  468.     }

  495.  
  496. /* Katsavounidis et al. "A New Initialization Technique for Generalized
  497.  * Lloyd Iteration", IEEE Signal Proc. Lett. 1 (10), 144-146, 1994. 
  498.  */
  499. function distributeOverClusters2($k, $arr)
  500. {
  501.   // set 1 minute timeout for dealing with large vectors
  502.   ini_set('max_execution_time', 60);
  503.   
  504.   $maxX = 0; $maxY = 0; $maxNorm = 0; $maxIndex = 0;
  505.   // init
  506.   foreach ($arr as $i => $point) 
  507.   {
  508.     if ($point->x > $maxX) { $maxX = $point->x; }
  509.     if ($point->y > $maxY) { $maxY = $point->y; }
  510.     // calculate the L2 norm of all mouse points (2D vector)
  511.     $norm = vectorNorm($point, "uL2");
  512.     if ($norm > $maxNorm) { 
  513.       $maxNorm = $norm;
  514.       $maxIndex = $i; 
  515.     }
  516.   }
  517.   
  518.   $clusters = array();
  519.   // choose the vector with the maximum norm as the first codeword
  520.   $clusters[0] = new Cluster($arr[$maxIndex]);
  521.   // check
  522.   if ($k < 2) {
  523.     $clusters->points = $arr;
  524.     return $clusters;
  525.   }
  526.   /* Calculate the distance of all mouse points from the first codeword
  527.    * and choose the vector with larger distance as the second codeword
  528.    */  
  529.   $clusters[1] = deployCluster($clusters[0]->avgPoint, $arr);
  530.   
  531.   /* With codebook of size > 2, compute the distance between any remaining
  532.    * vector and all existing codewords
  533.    */  
  534.   for ($i = 2; $i < $k; ++$i) {
  535.     $center = new Point(rand(0, $maxX), rand(0, $maxY));
  536.     $clusters[$i] = deployCluster($center, $arr);
  537.   }
  538.   for ($a = 0; $a < 5; ++$a)         // 5 iterations is enough...
  539.   {       
  540.     foreach ($clusters as $cluster) {
  541.       $cluster->points = array();     // reinitialize points
  542.     }
  543.     // compute best distance
  544.     foreach ($arr as $pnt) 
  545.     {
  546.       $bestcluster = $clusters[0];
  547.       $bestdist = $clusters[0]->avgPoint->getDistance($pnt);
  548.       
  549.       foreach ($clusters as $cluster) {
  550.         $distance = $cluster->avgPoint->getDistance($pnt);
  551.         if ($distance < $bestdist) {
  552.           $bestcluster = $cluster;
  553.           $bestdist = $distance;
  554.         }
  555.       }
  556.       // add the point to the best cluster.
  557.       $bestcluster->points[] = $pnt;
  558.     }
  559.     // recalculate the centers and sample variance
  560.     foreach ($clusters as $cluster) {
  561.       $cluster->calculateAverage($maxX, $maxY);
  562.       $cluster->calculateVariance();
  563.     } 
  564.   } // end loop 
  565.   return $clusters;
  566. }

  590. /** 
  591.  * Vector Norm calculation
  592.  * @param   array   $array  input Array
  593.  * @param   string  $type   norm calculation type: "L1", "uL2", "sL2", "Linf"
  594.  * @return  float           Vector Norm
  595.  * @link http://w3.gre.ac.uk/~physica/phy3.00/theory/node97.htm
  596.  */
  597. function vectorNorm($array, $type) 
  598. {
  599.   /* Note: mouse coordinates are positive values always, so we don't need 
  600.    * to compute the absolute values first for cases "sL2", "Linf" and "Linf2".
  601.    */
  602.   switch ($type) 
  603.   {
  604.     /** 1-norm */
  605.     case "L1":
  606.       foreach ($array as $value) { $sum += abs($value); }
  607.       $result = $sum;
  608.       break;
  609.         
  610.     /** Unscaled 2-norm */  
  611.     case "uL2":
  612.     default:
  613.       foreach ($array as $value) { $sum += $value*$value; }
  614.       $result = sqrt($sum);
  615.       break;
  616.       
  617.     /** Scaled 2-norm */
  618.     case "sL2":
  619.       $rmax = max($array);
  620.       foreach ($array as $value) {
  621.         $value /= $rmax; 
  622.         $sum += $value*$value; 
  623.       }
  624.       $result = $rmax * sqrt($sum);
  625.       break;
  626.       
  627.     /** Infinity norm */
  628.     case "Linf":
  629.       $result = max($array);
  630.       break;
  631.       
  632.     /** Negative infinity norm */
  633.     case "Linf2":
  634.       $result = min($array);
  635.       break;  
  636.   }
  637.   
  638.   return $result;
  639. }
