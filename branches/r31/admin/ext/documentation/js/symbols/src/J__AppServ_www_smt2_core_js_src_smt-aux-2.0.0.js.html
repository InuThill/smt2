 28   /**
 29    * Allows recording/replaying the mouse path over Flash objects.
 30    * A Flash movie may display above all the layers on the HTML apge, 
 31    * regardless of the stacking order ("z-index") of those layers.   
 32    * Using a WMODE value of "opaque" or "transparent" will prevent a Flash movie from playing in the topmost layer 
 33    * and allow you to adjust the layering of the movie within other layers of the HTML document.
 34    * However, to avoid possible performance issues, it's best use the "opaque" mode.   
 35    * Note: The WMODE parameter is supported only on some browser/Flash Player version combinations. 
 36    * If the WMODE parameter is not supported, the Flash movie will always display on top.
 37    * @return void  
 38    */
 39   allowTrackingOnFlashObjects: function()
 40   {
 41     var obj = document.getElementsByTagName("object");  
 42     for (var i = 0, t = obj.length; i < t; ++i) {
 43       var param = document.createElement("param");
 44       param.setAttribute("name", "wmode");
 45       param.setAttribute("value","opaque");
 46       obj[i].appendChild(param);
 47     }
 48     
 49     var embed = document.getElementsByTagName("embed");
 50     for (var j = 0, u = embed.length; j < u; ++j) {
 51       embed[j].setAttribute("wmode", "opaque");
 52       // recording on some browsers is tricky (replaying is ok, though)
 53       if (!/MSIE/i.test(navigator.userAgent)) {
 54         /* Some browsers sets the wmode correctly, 
 55          * but once the SWF object is instantiated you can't update its properties.
 56          * So, replace the old Flash object with the new one ;)       
 57          */
 58         var cloned = embed[j].cloneNode(true);
 59         embed[j].parentNode.replaceChild(cloned, embed[j]);
 60       }
 61     }

 90   /**
 91    * Checks the DOM-ready initialization in modern browsers.
 92    * This method was introduced by Dean Edwards/Matthias Miller/John Resig (dean.edwards.name/outofhanwell.com/jquery.com)
 93    * and it is discussed in http://dean.edwards.name/weblog/2006/06/again/
 94    * It's 2009, and fortunately nowadays there are only 2 types of modern browsers: W3C standards and Internet Explorer.  
 95    * @return void
 96    * @param {function} callback   the function to be called on DOM load
 97    */
 98   onDOMload: function(callback)
 99   {
100     if (arguments.callee.done) { return; }
101     // flag this function so we don't do the same thing twice
102     arguments.callee.done = true;
103     
104     // Firefox, Opera, Webkit-based browsers (Chrome, Safari)...
105     if (document.addEventListener) {
106       document.addEventListener('DOMContentLoaded', callback, false);
107       document.removeEventListener('DOMContentLoaded', callback, false);
108     }
109     // Internet Explorer ¬¬
110     else if (document.attachEvent) {
111       try {
112         document.write("<scr"+"ipt id=__ie_onload defer=true src=//:><\/scr"+"ipt>");
113       	var script = document.getElementById("__ie_onload"); 
114       	script.onreadystatechange = function() {
115           if (this.readyState === 'complete') { callback(); }
116       	};
117       } catch(e) {}
118     }
119     // fallback: old browsers use the window.onload event
120     this.addEvent(window, 'load', callback);

197   /**
198    * Gets the max value from both window (viewport's size) and document's size, adding offsets.
199    * MSHTML (Trident), Firefox (Gecko), Opera (Presto), and Safari (Webkit) support offsetHeight on document.body or on document.documentElement. 
200    * The difficulty is figuring out which one is reliable, because it depends on the environment. 
201    * As usual, the environment includes the browser, its version, and the rendering mode of the document.     
202    * @return {object} viewport dimmensions object with 2 properties: width {integer}, and height {integer}  
203    */
204   getPageSize: function()
205   {
206     var win = this.getWindowSize(),
207         doc = this.getDocumentSize();
208 
209     // find max values from this group
210     var w = (doc.width < win.width) ? win.width : doc.width;
211     var h = (doc.height < win.height) ? win.height : doc.height;
212     // however, some browsers in some situations should return more space available
213     var d = document.documentElement || document.body;
214     // verify offsets...
215     if (h < d.offsetHeight) { h = d.offsetHeight; }
216     if (w < d.offsetWidth) { w = d.offsetWidth; }
217     // ...and scrolls
218     if (h < d.scrollHeight) { h = d.scrollHeight; }
219     if (w < d.scrollWidth) { w = d.scrollWidth; }
220 
221     return { width: w, height: h };
222   },
223   /**
224    * Failsafe function. If your pages' HTML code does not allocate the image's dimensions (width/height attributes),
225    * then the page size will not be accurately computed and the tracking layer can be cropped.
226    * This occurs sometimes due to the replaying script begins on DOM load.   
227    * @return {object} viewport dimmensions object with 2 properties: width {integer}, and height {integer}  
228    */
229   getHardcorePageSize: function()
230   {
231     // let's define some helpers
232     var w1=0, h1=0, w2=0, h2=0, w3=0, h3=0, w4=0, h4=0,
233         de = document.documentElement,
234         db = document.body;
235   
236     if (window.innerHeight) {
237   		w1 = window.innerWidth;
238   		h1 = window.innerHeight;
239   		if (window.scrollMaxY) {
240   		  w1 += window.scrollMaxY;
241   		  h1 += window.scrollMaxY;
242       }
243   	}
244   	if (de.offsetHeight) {
245       w2 = de.offsetWidth;
246       h2 = de.offsetHeight;
247   	}
248   	if (db.scrollHeight) {
249   		w3 = db.scrollWidth;
250   		h3 = db.scrollHeight;
251   	}
252   	if (db.offsetWidth) {
253       w4 = db.offsetWidth;
254       h4 = db.offsetHeight;
255       if (db.offsetLeft) {
256         w4 += db.offsetLeft;
257         h4 += db.offsetTop;
258       }
259     }
260     // now return the max value
261   	var maxHeight = Math.max(h1,h2,h3,h4);
262   	var maxWidth  = Math.max(w1,w2,w3,w4);
263   	
264     return { width: maxWidth, height: maxHeight };
265   },
266   /**
267    * Gets the max z-index level available on the page.
268    * @return {integer}    z-index level     
269    * @param {object} e    DOM element (default: document)
270    * @autor Jason J. Jaeger (greengeckodesign.com)   
271    */
272   getNextHighestDepth: function(e) 
273   {
274     var highestIndex = 0;
275     var currentIndex = 0;
276     var elementArray = [];
277     // check all elements in page ...
278     if (document.getElementsByTagName) {
279       elementArray = document.getElementsByTagName('*');
280     } else if (e.getElementsByTagName) {
281       elementArray = document.getElementsByTagName('*');
282     }
283     // ... and iterate
284     for (var i = 0, l = elementArray.length; i < l; ++i) {
285       if (elementArray[i].currentStyle) {
286         currentIndex = parseFloat(elementArray[i].currentStyle.zIndex);
287       } else if (window.getComputedStyle) {
288         currentIndex = parseFloat(document.defaultView.getComputedStyle(elementArray[i],null).getPropertyValue('z-index'));
289       }
290       if (currentIndex > highestIndex) { highestIndex = currentIndex; }
291     }
292 
293     return highestIndex + 1;
294   },

373   /** 
374    * Makes an asynchronous XMLHTTP request (XHR) via GET or POST.
375    * Inspired on Peter-Paul Koch's XMLHttpRequest function.   
376    * @return void
377    * @param  {object}    reqSetup   Request properties
378    * @config {string}    url        Request URL
379    * @config {function} [callback]  Response function
380    * @config {string}   [postdata]  POST vars in the form "var1=name&var2=name..."
381    * @config {object}   [xmlhttp]   A previous XMLHTTP object can be reused

384   {
385     // create XHR object (or reuse it)
386     var request = (setup.xmlhttp) ? setup.xmlhttp : this.createXMLHTTPObject();
387     if (!request) { return; }
388     
389     var method = (setup.postdata) ? "POST" : "GET";
390     // start request
391     request.open(method, setup.url, true);
392     // post requests must set the correct content type
393     if (setup.postdata) {
394       request.setRequestHeader('Content-Type', "application/x-www-form-urlencoded");
395     }
396     // check for the 'complete' request state
397     request.onreadystatechange = function(){
398       if (this.readyState == 4 && typeof setup.callback === 'function') {
399         // send server response to callback function
400         setup.callback(request.responseText);
401       }
402     };
403     // send request
404     request.send(setup.postdata);

590     getClass: function(o) 
591     {
592       // do not track HTML and BODY nodes
593       if (o.nodeName == 'HTML' || o.nodeName == 'BODY') { return null; }
594       // if the element has no class, return its node name only
595       return (o.className) ? o.nodeName + "." + o.className.split(" ")[0] : o.nodeName;
596     },
597     /** 
598      * Gets all node parents until a node with ID is found.
599      * @return {string}     CSS selector string (intended to use in other scripts).
600      * @param {object}  o   DOM node element        
601      */   
602     getParents: function(o) 
603     {
604       // store current node first
605       var elem = (o.id) ? this.getID(o) : this.getClass(o);
606       var list = [elem]; //(elem) ? [elem] : [];
607       // get all parents until find an element with ID
608       while (o.parentNode) {
609         o = o.parentNode;
610         // store only element nodes
611         if (o.nodeType == 1) {
612           if (o.id) {
613             elem = this.getID(o);
614             list.unshift(elem);
615             // if parent has an ID, end finding
616             return list.join(" ");
617           } else {
618             elem = this.getClass(o);
619             list.unshift(elem);
620           } 
621           if (o == parent) {
622             // #document reached
623             return list.join(" ");
624           }
625         }
626       }
627       return list.join(" ");
628     }
629   },
630   /**
631    * Array methods -- without extending the Array prototype (best practice). 
632    * Note: These Array methods would only work for a completely 'dense' array. 
633    * If you're working with sparse arrays, then you should pre-process them:
634    * <code> 
635    *   var narray = [];
636    *   for (var i = 0; i < array.length; ++i) {
637    *     if (array[i] != null) { narray.push(array[i]); }
638    *   }
639    *   // now narray is converted in a 'dense' array
640    * </code>   
641    */
642   array: {
643     /**
644      * Gets the max element in a numeric array. 
645      * @return {float}        array max value
646      * @param {array} array 
647      * @author John Resig
648      */
649     max: function(array) 
650     {
651       return Math.max.apply(Math, array);
652     },
653     /**
654      * Gets the min element in a numeric array.
655      * @return {float}        array min value
656      * @param {array} array 
657      * @author John Resig
658      */
659     min: function(array) 
660     {
661       return Math.max.apply(Math, array);
662     },
663     /**
664      * Sums all elements in a numeric array.
665      * @return {float}        array sum
666      * @param {array} array   
667      */
668     sum: function(array) 
669     {
670       var s = 0;
671       for (var i = 0, l = array.length; i < l; ++i) {
672         s += array[i];
673       }
674       return s; 
675     },
676     /**
677      * Appends any supplied arguments to the end of the array, in the order given.
678      * @return {integer}        new length of the array
679      * @param {array} array
680      * @param {mixed} content   anything to push to array  
681      * @author Ash Searle (tweaked by Luis Leiva)
682      * @link http://hexmen.com/blog/ 
683      * @license http://creativecommons.org/licenses/by/2.5/
684      */
685     push: function(array, content) 
686     {
687       var args = Array.prototype.slice.call(content);
688       var n = array.length >>> 0;
689       for (var i = 0; i < args.length; ++i) {
690     	 array[n] = args[i];
691     	 n = n + 1 >>> 0;
692       }
693       array.length = n;
694       return n;
695     },
696     /**
697      * Removes the last element of the array and returns it.
698      * @return {mixed}        element removed (if any). Otherwise return <code>undefined</code>
699      * @param {array} array 
700      * @author Ash Searle
701      * @link http://hexmen.com/blog/ 
702      * @license http://creativecommons.org/licenses/by/2.5/
703      */
704     pop: function(array) 
705     {
706       var n = array.length >>> 0, value;
707       if (n) {
708       	value = array[--n];
709       	delete array[n];
710       }
711       array.length = n;
712       return value;
713     },
714     /**
715      * Removes element(s) in desired position(s) from input array and returns the modified array.
716      * @return {array}          modified array
717      * @param {array}   array
718      * @param {integer} from    start index
719      * @param {integer} to      end index
720      * @author John Resig 
721      */
722     remove: function(array, from, to) 
723     {
724       var rest = array.slice((to || from) + 1 || array.length);
725       array.length = (from < 0) ? array.length + from : from;
726       return array.push.apply(array, rest);
727     },
728     /**
729      * Gets the index of the first element that matches search value.
730      * @return {integer}          index if found, or -1 otherwise
731      * @param {array}   array
732      * @param {mixed}   search    element to search for
733      * @param {integer} from      start index      
734      * @link http://snipplr.com/view/3355/arrayindexof/   
735      */
736     indexOf: function(array, search, from) 
737     {
738       for (var i = (from || 0), total = array.length; i < total; ++i) {
739         // strict types should use === comparison
740         if (array[i] == search) {
741           return i;
742         }
743       }
744       return -1;
745     }
746   }
